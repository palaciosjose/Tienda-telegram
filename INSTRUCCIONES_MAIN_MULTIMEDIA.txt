# Obtener información multimedia del producto
media_info = dop.get_product_media(callback.data)

if media_info:
    # El producto tiene multimedia
    try:
        # Enviar multimedia con información del producto
        if media_info['type'] == 'photo':
            bot.edit_message_media(
                chat_id=callback.message.chat.id,
                message_id=callback.message.message_id,
                media=telebot.types.InputMediaPhoto(
                    media=media_info['file_id'],
                    caption=dop.format_product_with_media(callback.data)
                ),
                reply_markup=key
            )
        elif media_info['type'] == 'video':
            bot.edit_message_media(
                chat_id=callback.message.chat.id,
                message_id=callback.message.message_id,
                media=telebot.types.InputMediaVideo(
                    media=media_info['file_id'],
                    caption=dop.format_product_with_media(callback.data)
                ),
                reply_markup=key
            )
        else:
            # Para documentos, audio, etc., eliminar mensaje anterior y enviar nuevo
            bot.delete_message(callback.message.chat.id, callback.message.message_id)
            
            if media_info['type'] == 'document':
                bot.send_document(
                    chat_id=callback.message.chat.id,
                    document=media_info['file_id'],
                    caption=dop.format_product_with_media(callback.data),
                    reply_markup=key,
                    parse_mode='Markdown'
                )
            elif media_info['type'] == 'audio':
                bot.send_audio(
                    chat_id=callback.message.chat.id,
                    audio=media_info['file_id'],
                    caption=dop.format_product_with_media(callback.data),
                    reply_markup=key,
                    parse_mode='Markdown'
                )
    except:
        # Si falla enviar multimedia, usar mensaje de texto
        bot.edit_message_text(
            chat_id=callback.message.chat.id,
            message_id=callback.message.message_id,
            text=dop.format_product_with_media(callback.data),
            reply_markup=key,
            parse_mode='Markdown'
        )
else:
    # Producto sin multimedia, usar mensaje de texto normal
    bot.edit_message_text(
        chat_id=callback.message.chat.id,
        message_id=callback.message.message_id,
        text=dop.format_product_with_media(callback.data),
        reply_markup=key,
        parse_mode='Markdown'
    )
